# 类型提示详解

> Python 3.5+ 引入了类型提示。它不是运行时强制的，但能帮助 IDE、静态检查工具（mypy）和你自己理解代码。本章完整讲解类型提示的语法和最佳实践。

---

## 1. 为什么需要类型提示

### 没有类型提示的代码

```python
def process(data, config):
    """data 是什么类型？config 是什么？返回什么？"""
    result = []
    for item in data:
        if item > config.get("threshold"):
            result.append(item * 2)
    return result

# 调用者只能靠猜测
process([1, 2, 3], {"threshold": 1})
```

### 有类型提示的代码

```python
def process(data: list[int], config: dict[str, int]) -> list[int]:
    """一目了然：接收整数列表和配置字典，返回整数列表"""
    result = []
    for item in data:
        if item > config.get("threshold", 0):
            result.append(item * 2)
    return result

# IDE 能自动补全、能检查类型错误
```

## 2. 基本类型注解

### Python 3.9+ 内置泛型语法

```python
# 基本类型
name: str = "Alice"
age: int = 30
price: float = 9.99
is_active: bool = True
nothing: None = None

# 容器类型（Python 3.9+ 直接用内置类型）
scores: list[int] = [95, 88, 76]
coordinates: tuple[float, float] = (3.14, 2.71)
config: dict[str, int] = {"threshold": 5, "timeout": 30}
unique_ids: set[int] = {1, 2, 3}

# Python 3.8 及以下需要使用 typing 模块
from typing import List, Dict, Tuple, Set
scores: List[int] = [95, 88, 76]       # 旧写法
config: Dict[str, int] = {"threshold": 5}
```

### 函数类型注解

```python
def greet(name: str) -> str:
    return f"Hello, {name}!"

def divide(a: float, b: float) -> float:
    if b == 0:
        raise ValueError("除数不能为零")
    return a / b

# 无返回值
def log(message: str) -> None:
    print(f"[LOG] {message}")

# 永不返回（如 sys.exit 或无限循环）
from typing import NoReturn

def fatal_error(msg: str) -> NoReturn:
    raise SystemExit(msg)
```

## 3. 高级类型

### Optional — 可能为 None

```python
# Optional[X] 等价于 X | None
def find_user(user_id: int) -> str | None:       # Python 3.10+
    if user_id > 0:
        return "found"
    return None

# 旧写法
from typing import Optional
def find_user(user_id: int) -> Optional[str]:     # Python 3.9-
    ...
```

### Union — 多种类型之一

```python
# X | Y 等价于 Union[X, Y]（Python 3.10+）
def parse(value: int | str | float) -> float:
    return float(value)

# 旧写法
from typing import Union
def parse(value: Union[int, str, float]) -> float:
    return float(value)
```

### Any — 任意类型

```python
from typing import Any

def deserialize(data: str) -> Any:
    """返回类型不确定"""
    import json
    return json.loads(data)

result = deserialize('{"name": "Alice"}')
# result 的类型是 Any，IDE 不会提示
```

**原则**：尽量避免 Any。能用具体类型就用具体类型。

### Literal — 限定值

```python
from typing import Literal

def set_mode(mode: Literal["read", "write", "append"]) -> None:
    """mode 只能是这三个字符串之一"""
    print(f"设置模式: {mode}")

set_mode("read")    # OK
# set_mode("delete")  # IDE 会警告！
```

### Callable — 函数类型

```python
from typing import Callable

# def apply(func: Callable[[int, int], int], a: int, b: int) -> int:
#   Callable[[参数类型列表], 返回值类型]
def apply(func: Callable[[int, int], int], a: int, b: int) -> int:
    return func(a, b)

def add(x: int, y: int) -> int:
    return x + y

print(apply(add, 3, 5))  # 8
```

## 4. 容器类型详解

```python
# 列表
names: list[str] = ["Alice", "Bob"]
matrix: list[list[int]] = [[1, 2], [3, 4]]

# 字典
scores: dict[str, int] = {"math": 95, "english": 88}
nested: dict[str, dict[str, int]] = {
    "user1": {"math": 95, "english": 88}
}

# 元组
# 固定长度的元组可以指定每个位置类型
point: tuple[float, float, float] = (1.0, 2.0, 3.0)
# 可变长度的元组
records: tuple[int, ...] = (1, 2, 3, 4, 5)

# 集合
tags: set[str] = {"python", "tutorial"}

# 序列（更通用的类型）
from typing import Sequence
def process(items: Sequence[int]) -> int:
    """接收任何序列：list, tuple, range 等"""
    return sum(items)

# 映射（更通用的字典类型）
from typing import Mapping
def lookup(d: Mapping[str, int], key: str) -> int:
    return d.get(key, 0)
```

## 5. TypeVar 泛型

```python
from typing import TypeVar

T = TypeVar('T')  # 泛型类型变量

def first(items: list[T]) -> T | None:
    """返回类型与列表元素类型相同"""
    return items[0] if items else None

print(first([1, 2, 3]))     # 返回 int | None
print(first(["a", "b"]))    # 返回 str | None

# 限制 TypeVar
Number = TypeVar('Number', int, float)  # 只能是 int 或 float

def double(x: Number) -> Number:
    return x * 2

print(double(5))    # 10
print(double(3.14)) # 6.28
# double("hi")      # IDE 警告！
```

## 6. TypedDict — 类型化的字典

```python
from typing import TypedDict

class UserDict(TypedDict):
    """定义字典的键和值类型"""
    name: str
    age: int
    email: str

def create_user(data: UserDict) -> str:
    return f"{data['name']} ({data['age']})"

# 使用
user: UserDict = {
    "name": "Alice",
    "age": 30,
    "email": "alice@example.com"
}
print(create_user(user))  # Alice (30)
```

### 可选键和只读键

```python
from typing import TypedDict

class Config(TypedDict, total=False):
    """total=False 表示所有键都是可选的"""
    debug: bool
    timeout: int
    host: str

config: Config = {"debug": True}  # 只提供部分键，OK

# 或者单独标记
class User(TypedDict):
    name: str          # 必需
    age: int           # 必需
    nickname: str      # 可选（使用 NotRequired，Python 3.11+）
```

## 7. Protocol — 结构化子类型（鸭子类型）

```python
from typing import Protocol

class Flyable(Protocol):
    """任何有 fly 方法的类都满足这个协议"""
    def fly(self) -> str:
        ...

class Bird:
    def fly(self) -> str:
        return "鸟在飞"

class Airplane:
    def fly(self) -> str:
        return "飞机在飞"

def make_it_fly(thing: Flyable) -> str:
    return thing.fly()

make_it_fly(Bird())      # OK
make_it_fly(Airplane())  # OK — 不需要显式继承 Flyable！
# Protocol 实现了"鸭子类型"

class Rock:
    pass

# make_it_fly(Rock())    # IDE 警告：Rock 没有 fly 方法
```

## 8. Final 和常量

```python
class BaseConfig:

    @final
    def start(self) -> None:
```

## 9. 类自身的类型注解

```python
class ListNode:
    def __init__(self, value: int, next_node: "ListNode | None" = None):
        self.value = value
        self.next = next_node  # 引用自身类型，用字符串

# Python 3.11+ 可以用 typing.Self
from typing import Self

class Builder:
    def set_name(self, name: str) -> Self:
        self.name = name
        return self  # 返回自身，支持链式调用

    def set_age(self, age: int) -> Self:
        self.age = age
        return self

    def build(self) -> "User":
        return User(self.name, self.age)

# 链式调用
user = Builder().set_name("Alice").set_age(30).build()
```

## 10. 类型别名

```python
# 让复杂类型更可读
type UserId = int                              # Python 3.12+
type JsonDict = dict[str, "JsonValue"]         # 递归别名
type JsonValue = str | int | float | bool | None | list["JsonValue"] | "JsonDict"

# 旧版本
UserId = int
Vector = tuple[float, float, float]
Callback = Callable[[int, str], None]
```

## 11. mypy 静态类型检查

### 安装和使用

```bash
pip install mypy
mypy script.py
```

### 示例

```python
# script.py
def add(a: int, b: int) -> int:
    return a + b

result = add(1, "2")  # ← 类型错误！
```

```bash
$ mypy script.py
script.py:5: error: Argument 2 to "add" has incompatible type "str"; expected "int"
```

### 配置 mypy

```ini
# mypy.ini 或 pyproject.toml

python_version = "3.12"
strict = true
warn_return_any = true
warn_unused_configs = true
```

## 12. dataclass 类型注解

```python
from dataclasses import dataclass, field

@dataclass
class User:
    name: str
    age: int
    email: str = ""
    tags: list[str] = field(default_factory=list)
    active: bool = True

    def greeting(self) -> str:
        return f"Hello, I'm {self.name}"

# 自动生成 __init__、__repr__、__eq__
user = User(name="Alice", age=30, tags=["admin", "user"])
print(user)  # User(name='Alice', age=30, email='', tags=['admin', 'user'], active=True)
```

## 13. 类型提示的最佳实践

```python
# DO ✅
def get_user(user_id: int) -> User | None:
    ...

def calculate(items: Sequence[float]) -> float:
    ...

# DON'T ❌
def get_user(user_id) -> object:      # 太宽泛
    ...

def calculate(items) -> float:        # 参数无类型
    ...

# 原则：
# 1. 所有公共函数都要有类型注解
# 2. 尽量用具体类型，避免 Any
# 3. 复杂类型提取别名
# 4. 用 Protocol 代替显式 ABC 继承
# 5. 运行 CI 中集成 mypy 检查
```

## 14. 小结

| 语法 | 说明 | 示例 |
|------|------|------|
| `x: int` | 基本类型 | `age: int = 30` |
| `-> str` | 返回值类型 | `def f() -> str:` |
| `X \| Y` | 联合类型 | `int \| str` |
| `list[X]` | 泛型容器 | `list[int]` |
| `Optional[X]` | 可为 None | `str \| None` |
| `Callable` | 函数类型 | `Callable[[int], str]` |
| `TypeVar` | 泛型变量 | `T = TypeVar('T')` |
| `Protocol` | 结构化子类型 | 鸭子类型 |
| `TypedDict` | 类型化字典 | 定义字典结构 |
| `Literal` | 限定值 | `Literal["a", "b"]` |
| `Final` | 常量/不可重写 | `MAX: Final = 100` |
