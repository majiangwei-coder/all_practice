# 常用内置函数速查

> Python 有 70+ 内置函数。本章精选最常用的，按分类整理，方便查阅。

---

## 1. 类型转换

```python
# 数字转换
int("42")         # 42
int("ff", 16)     # 255 (指定进制)
float("3.14")     # 3.14
complex(1, 2)     # (1+2j)
bool(1)           # True
bool([])          # False

# 字符串转换
str(42)           # "42"
repr("hello")     # "'hello'" (带引号)
chr(65)           # 'A'   (Unicode → 字符)
ord('A')          # 65    (字符 → Unicode)
hex(255)          # '0xff'
oct(8)            # '0o10'
bin(5)            # '0b101'

# 容器转换
list("abc")       # ['a', 'b', 'c']
tuple([1,2,3])    # (1, 2, 3)
dict([(1,'a'),(2,'b')])  # {1:'a', 2:'b'}
set([1,2,2,3])    # {1, 2, 3}
frozenset([1,2])  # frozenset({1, 2}) - 不可变集合
bytes([104,101])  # b'hello'
bytearray(b'hi')  # bytearray(b'hi') - 可变字节

# ASCII 字符转换
ord('A')          # 65
chr(65)           # 'A'
```

## 2. 数学运算

```python
abs(-5)           # 5  (绝对值)
round(3.14159, 2) # 3.14 (四舍五入)
pow(2, 3)         # 8  (等价于 2**3)
pow(2, 3, 5)     # 3  (2**3 % 5)
divmod(10, 3)    # (3, 1) (商, 余数)

sum([1, 2, 3])   # 6
min(3, 1, 4)      # 1
max(3, 1, 4)      # 4
min([3,1,4], key=abs)  # 1
```

## 3. 迭代与序列

```python
# range
range(5)          # 0, 1, 2, 3, 4
range(2, 5)       # 2, 3, 4
range(0, 10, 2)   # 0, 2, 4, 6, 8

# enumerate — 同时获取索引和值
for i, v in enumerate(['a','b','c']):
    print(i, v)   # 0 a, 1 b, 2 c

enumerate(['a','b','c'], start=1)  # (1,'a'), (2,'b'), (3,'c')

# zip — 并行遍历
list(zip([1,2,3], ['a','b','c']))  # [(1,'a'), (2,'b'), (3,'c')]
list(zip([1,2], ['a','b','c']))    # [(1,'a'), (2,'b')] (短者截断)

# zip 配合 strict (Python 3.10+)
# list(zip([1,2], ['a','b','c'], strict=True))  # ValueError! 长度不同

# reversed — 反向迭代
list(reversed([1,2,3]))  # [3, 2, 1]

# sorted — 排序
sorted([3, 1, 2])               # [1, 2, 3]
sorted([3, 1, 2], reverse=True)  # [3, 2, 1]
sorted(['apple','hi','banana'], key=len)  # ['hi', 'apple', 'banana']

# filter — 过滤
list(filter(lambda x: x > 2, [1,2,3,4]))   # [3, 4]
list(filter(None, [0, 1, '', 'a', False])) # [1, 'a']

# map — 映射
list(map(str, [1, 2, 3]))        # ['1', '2', '3']
list(map(pow, [2,3], [4,5]))    # [16, 243] (2**4, 3**5)
```

## 4. 属性与类型检查

```python
# type
type(42)          # <class 'int'>
type("hello")     # <class 'str'>

# isinstance — 类型检查
isinstance(42, int)           # True
isinstance(42, (int, float))  # True
isinstance(True, int)         # True (bool 是 int 子类)

# issubclass — 继承关系
issubclass(bool, int)         # True

# hasattr / getattr / setattr / delattr
class Obj:
    x = 1

hasattr(Obj, 'x')       # True
getattr(Obj, 'x')       # 1
getattr(Obj, 'y', 0)    # 0 (默认值)
setattr(Obj, 'y', 2)

# dir — 列出属性和方法
dir([])   # ['append', 'clear', 'copy', ...]

# id — 内存地址
id(42)          # 一串数字

# callable — 是否可调用
callable(print)  # True
callable(42)     # False
```

## 7. 高阶函数

```python
# map — 对每个元素应用函数
list(map(lambda x: x**2, [1,2,3]))   # [1, 4, 9]

# filter — 筛选满足条件的元素
list(filter(lambda x: x > 0, [-1, 0, 1, 2]))  # [1, 2]

# 注意：map/filter 返回迭代器，需要 list() 转换
# 很多时候列表推导式更 Pythonic:
[x**2 for x in [1,2,3]]            # 代替 map
[x for x in [-1,0,1,2] if x > 0]   # 代替 filter
```

## 8. 对象信息

```python
# help — 查看帮助
help(str)         # str 类的完整文档
help(str.upper)   # upper 方法的文档

# dir — 列出成员
dir([])           # 列表的所有方法和属性

# vars — 查看 __dict__
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

p = Point(3, 5)
vars(p)           # {'x': 3, 'y': 5} (实例属性)
vars(Point)       # 类的属性和方法

# id — 唯一标识
id(p)             # 内存地址
```

## 9. 动态执行

```python
# eval — 执行表达式（慎用！）
eval("2 + 3")     # 5
eval("[1, 2, 3]")  # [1, 2, 3]

# exec — 执行代码块（慎用！）
exec("x = 1\ny = 2\nprint(x + y)")  # 3

# compile — 编译代码
code = compile("a + b", "<string>", "eval")
eval(code, {"a": 1, "b": 2})  # 3

# ⚠️ 安全警告：永远不要对不可信输入使用 eval/exec！
```

## 10. 其他常用

```python
# len — 长度
len([1, 2, 3])    # 3
len("hello")      # 5
len({"a": 1})     # 1

# any / all
any([False, False, True])  # True (有一个为真)
all([True, True, True])    # True (全部为真)
all([True, False])         # False

# bytes 与 bytearray
bytes([65, 66])        # b'AB'
bytearray(b'hello')   # 可变的 bytes

# memoryview — 内存视图（不复制）
data = bytearray(b'hello')
mv = memoryview(data)
mv[0]     # 104

# classmethod / staticmethod / property — 见面向对象章节

# super — 调用父类方法
class Child(Parent):
    def method(self):
        super().method()

# __import__ — 动态导入
math = __import__("math")
math.sqrt(16)     # 4.0
# 推荐用 importlib.import_module 代替
```

## 12. 小结

- 多用 `enumerate` 而非 `range(len(...))`
- 多用列表推导式而非 `map`/`filter`
- `isinstance` 优于 `type() == `
- `any`/`all` 用于批量条件检查
- 永远不要对用户输入使用 `eval`/`exec`
