# 正则表达式

## re 模块基础

```python
import re

# 匹配
re.match(pattern, string)    # 从字符串开头匹配
re.search(pattern, string)   # 搜索第一个匹配
re.fullmatch(pattern, string) # 完全匹配整个字符串

# 查找全部
re.findall(pattern, string)  # 返回所有匹配的列表
re.finditer(pattern, string) # 返回所有匹配的迭代器

# 替换/分割
re.sub(pattern, repl, string)  # 替换
re.split(pattern, string)     # 分割
```

## 简单示例

```python
import re

text = "我的邮箱是 alice@example.com，另一个是 bob@test.org"

# search: 找第一个匹配
match = re.search(r'\w+@\w+\.\w+', text)
if match:
    print(match.group())  # alice@example.com

# findall: 找所有匹配
emails = re.findall(r'\w+@\w+\.\w+', text)
print(emails)  # ['alice@example.com', 'bob@test.org']

# sub: 替换
masked = re.sub(r'\w+@\w+\.\w+', '[EMAIL]', text)
print(masked)  # 我的邮箱是 [EMAIL]，另一个是 [EMAIL]
```

## 字符类

```python
[abc]     # 匹配 a, b, 或 c
[^abc]    # 匹配除了 a, b, c 的任意字符
[a-z]     # 匹配 a 到 z（小写字母）
[A-Z]     # 匹配 A 到 Z
[0-9]     # 匹配数字
[a-zA-Z0-9]  # 匹配字母数字
```

### 预定义字符类

| 符号 | 含义 | 等价 |
|------|------|------|
| `\d` | 数字 | `[0-9]` |
| `\D` | 非数字 | `[^0-9]` |
| `\w` | 单词字符 | `[a-zA-Z0-9_]` |
| `\W` | 非单词字符 | `[^a-zA-Z0-9_]` |
| `\s` | 空白字符 | `[ \t\n\r\f\v]` |
| `\S` | 非空白字符 | `[^ \t\n\r\f\v]` |
| `.` | 任意字符（除换行） | |

## 量词

```python
*      # 0 或多次（贪婪）
+      # 1 或多次（贪婪）
?      # 0 或 1 次
{3}    # 精确 3 次
{3,5}  # 3 到 5 次
{3,}   # 3 次或更多
```

### 贪婪 vs 非贪婪

```python
text = "<div>content</div>"

# 贪婪（默认）：尽可能多匹配
re.search(r'<.+>', text).group()  # '<div>content</div>'

# 非贪婪：尽可能少匹配
re.search(r'<.+?>', text).group()  # '<div>'
```

## 边界

```python
^       # 字符串开头（或行开头 in multiline）
$       # 字符串结尾（或行结尾 in multiline）
\b      # 单词边界
\B      # 非单词边界
```

```python
# 单词边界
text = "I like Python, python, PYTHON"
re.findall(r'\bpython\b', text, re.IGNORECASE)  # ['Python', 'python', 'PYTHON']

# 精确匹配开头
re.match(r'^hello', "hello world")  # 匹配
```

## 分组

```python
text = "张三的手机是138-1234-5678，李明的是139-9999-8888"

# 无分组
re.findall(r'\d{3}-\d{4}-\d{4}', text)
# ['138-1234-5678', '139-9999-8888']

# 有分组：返回分组内容
pattern = r'(\d{3})-(\d{4})-(\d{4})'
matches = re.findall(pattern, text)
# [('138', '1234', '5678'), ('139', '9999', '8888')]

# 命名分组
pattern = r'(?P<area>\d{3})-(?P<number>\d{4}-\d{4})'
match = re.search(pattern, text)
print(match.group('area'))    # '138'
print(match.group('number'))   # '1234-5678'
```

## 非捕获组

```python
text = "hello123hello456"

# 非捕获组：(?:...)
re.findall(r'(?:\d+)', text)  # ['123', '456']

# 捕获组：(\d+)
re.findall(r'(\d+)', text)    # ['123', '456']
# findall 行为相同，但 search 返回值不同
```

## 特殊字符转义

```python
# 匹配特殊字符需要转义
# re.escape() 会将所有正则元字符转义为字面字符
re.escape('file.txt')  # 'file\\.txt' (点号被转义)
re.escape('hello(world)')  # 'hello\\(world\\)'

# 在正则表达式中直接匹配特殊字符
re.search(r'\.', 'file.txt')  # 匹配点号
re.search(r'\(', 'hello(world)')  # 匹配左括号
```

## match 对象

```python
text = "电话号码：138-1234-5678"
match = re.search(r'(\d{3})-(\d{4})-(\d{4})', text)

if match:
    print(match.group())      # '138-1234-5678'（整体）
    print(match.group(1))     # '138'（第一个分组）
    print(match.group(2))     # '1234'
    print(match.group(3))     # '5678'
    print(match.groups())     # ('138', '1234', '5678')
    print(match.span())       # (5, 18)（位置）
    print(match.start())      # 5
    print(match.end())        # 18
```

## re.sub 的高级用法

```python
# 替换时使用分组
text = "2024-03-15"
result = re.sub(r'(\d{4})-(\d{2})-(\d{2})', r'\3/\2/\1', text)
print(result)  # 15/03/2024

# 替换函数
def add_prefix(match):
    return f"[IMG]{match.group()}[/IMG]"

text = "logo.png 和 banner.jpg"
result = re.sub(r'\w+\.(png|jpg)', add_prefix, text)
print(result)  # [IMG]logo.png[/IMG] 和 [IMG]banner.jpg[/IMG]
```

## re.split

```python
# 按模式分割
text = "apple,banana;cherry|orange"
re.split(r'[;,|]', text)  # ['apple', 'banana', 'cherry', 'orange']

# 保留分割符
parts = re.split(r'(\s+)', "hello   world")
# ['hello', '   ', 'world']
```

## compile 编译

```python
# 预编译正则表达式，提高多次使用时的性能
pattern = re.compile(r'\d{3}-\d{4}-\d{4}')

pattern.findall(text)     # 查找全部
pattern.search(text)      # 搜索第一个
pattern.sub('[PHONE]', text)  # 替换
```

## flags 选项

```python
# 多行模式
re.search(r'^hello', "hello\nworld\nhello", re.MULTILINE)

# 忽略大小写
re.findall(r'hello', "Hello HELLO hello", re.IGNORECASE)

# 点号匹配换行
re.search(r'hello.world', "hello\nworld", re.DOTALL)

# 组合
re.search(r'hello', "HELLO", re.IGNORECASE | re.MULTILINE)
```

## 实际应用

### 1. 验证输入

```python
import re

def validate_phone(phone):
    return bool(re.fullmatch(r'1[3-9]\d{9}', phone))

def validate_email(email):
    return bool(re.fullmatch(r'\w+@\w+\.\w+', email))

print(validate_phone("13812345678"))  # True
print(validate_phone("12345678901"))   # False
```

### 2. 提取结构化数据

```python
text = """
产品：iPhone 15，价格：6999，数量：2
产品：MacBook Pro，价格：19999，数量：1
"""

pattern = r'产品：(.+)，价格：(\d+)，数量：(\d+)'
for match in re.finditer(pattern, text):
    product, price, qty = match.groups()
    print(f"{product}: ¥{price} x {qty}")
```

### 3. 文本清洗

```python
import re

# 移除 HTML 标签
text = "<p>Hello <b>World</b>!</p>"
clean = re.sub(r'<[^>]+>', '', text)
print(clean)  # Hello World!

# 规范化空白
text = "Hello\s+World\t\nTest"
clean = re.sub(r'\s+', ' ', text)
print(clean)  # Hello World Test

# 提取数字
text = "价格是1999元，促销价仅999元"
numbers = re.findall(r'\d+', text)
print(numbers)  # ['1999', '999']
```

## 实战：常见解析场景

### 1. 解析日志文件

```python
import re

log_line = '2024-03-15 10:30:45,123 [ERROR] [main.py:42] 连接超时: db.example.com'

pattern = r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}),\d+ \[(\w+)\] \[([^\]]+)\] (.+)'
match = re.search(pattern, log_line)
if match:
    timestamp, level, location, message = match.groups()
    print(f"时间: {timestamp}")      # 2024-03-15 10:30:45
    print(f"级别: {level}")          # ERROR
    print(f"位置: {location}")       # main.py:42
    print(f"消息: {message}")       # 连接超时: db.example.com
```

### 2. 提取 URL 参数

```python
url = "https://example.com/search?q=python&page=2&size=20"

# 提取 query string
params = dict(re.findall(r'([^&=?]+)=([^&]*)', url))
print(params)  # {'q': 'python', 'page': '2', 'size': '20'}

# 验证 URL
url_pattern = r'^https?://[\w.-]+(:\d+)?(/[\w./-]*)?(\?[\w=&%-]*)?$'
print(bool(re.match(url_pattern, url)))  # True
```

### 3. 密码强度验证

```python
def password_strength(password):
    """检查密码强度"""
    checks = {
        "至少8个字符": len(password) >= 8,
        "包含大写字母": bool(re.search(r'[A-Z]', password)),
        "包含小写字母": bool(re.search(r'[a-z]', password)),
        "包含数字": bool(re.search(r'\d', password)),
        "包含特殊字符": bool(re.search(r'[!@#$%^&*(),.?":{}|<>]', password)),
    }
    return {k: v for k, v in checks.items()}

result = password_strength("Hello123!")
# {'至少8个字符': True, '包含大写字母': True, ...}
```

### 4. Markdown 到纯文本

```python
def strip_markdown(text):
    """去除 Markdown 格式标记"""
    # 去除链接: [text](url) → text
    text = re.sub(r'\[([^\]]+)\]\([^)]+\)', r'\1', text)
    # 去除加粗: **text** → text
    text = re.sub(r'\*\*([^*]+)\*\*', r'\1', text)
    # 去除斜体: *text* → text
    text = re.sub(r'\*([^*]+)\*', r'\1', text)
    # 去除行内代码: `code` → code
    text = re.sub(r'`([^`]+)`', r'\1', text)
    return text

md = "这是 **重要** 的 [链接](http://example.com) 和 `代码`"
print(strip_markdown(md))  # 这是 重要 的 链接 和 代码
```

## 常见的正则陷阱

### 回溯灾难（Catastrophic Backtracking）

```python
# 危险的模式：(a+)+b
# 对 "aaaaaaaaaaaaaaaaaaaaac" 这样的字符串
# 会导致指数级回溯，CPU 100%
# 解决：用原子组 (?>...) 或减少嵌套量词

# 不要这样写：
# re.match(r'(a+)+c', 'a' * 100 + 'b')  # 极慢！

# 应该这样写：
re.match(r'a+c', 'a' * 100 + 'b')  # 很快
```

### re.match vs re.search

```python
# re.match 只能从开头匹配
print(re.match(r'\d+', 'abc123'))  # None（不是以数字开头）
print(re.search(r'\d+', 'abc123'))  # <match object>（在中间找到了）

# 建议：始终用 re.search，除非明确要求从开头匹配
# 或用 re.fullmatch 要求完全匹配
```

### 原始字符串 r''

```python
# 正则中 \b 表示单词边界，但 Python 字符串的 \b 是退格符
# 所以正则表达式必须用原始字符串！

# 错误：
# re.search('\bword\b', text)   # \b 被 Python 解释为退格

# 正确：
re.search(r'\bword\b', text)   # r'' 让 \b 保持原样传给正则引擎
```

## 小结

1. **字符类**：`\d` 数字, `\w` 单词, `\s` 空白, `.` 任意
2. **量词**：`*` 0+, `+` 1+, `?` 0/1, `{n,m}` 范围
3. **分组**：`()` 捕获分组, `(?:)` 非捕获, `(?P<name>)` 命名
4. **边界**：`^` 开始, `$` 结束, `\b` 单词边界
5. **match 对象**：`group()`, `groups()`, `span()`
6. **re.sub**：替换，支持分组引用 `\1` 和替换函数
7. **compile**：预编译提高性能
8. **flags**：`IGNORECASE`, `MULTILINE`, `DOTALL`
9. **始终用 r'' 原始字符串**写正则模式
10. **注意回溯灾难**：避免嵌套量词
