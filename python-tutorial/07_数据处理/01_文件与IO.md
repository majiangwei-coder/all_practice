# 文件与 IO

## 打开文件：open()

```python
# 基础语法
open(file, mode='r', buffering=-1, encoding=None)

# 模式
'r'  # 读（默认）
'w'  # 写（覆盖）
'a'  # 追加
'x'  # 创建并写（文件存在则报错）

# 二进制模式
'rb'  # 读二进制
'wb'  # 写二进制
'ab'  # 追加二进制

# 读写模式
'r+'  # 读写（文件必须存在）
'w+'  # 读写（覆盖）
'a+'  # 读写（追加）
```

## 读取文件

```python
# 读取全部
with open("file.txt", "r", encoding="utf-8") as f:
    content = f.read()

# 读取一行
with open("file.txt", "r") as f:
    first_line = f.readline()

# 读取所有行
with open("file.txt") as f:
    lines = f.readlines()  # 列表

# 遍历（内存高效）
for line in open("file.txt", "r"):
    print(line, end="")
```

## 写入文件

```python
# 写入（覆盖）
with open("output.txt", "w", encoding="utf-8") as f:
    f.write("Hello\n")
    f.write("World\n")

# 追加
with open("output.txt", "a") as f:
    f.write("More content\n")

# 写入多行
lines = ["line1\n", "line2\n", "line3\n"]
with open("output.txt", "w") as f:
    f.writelines(lines)
```

## 最佳实践：with 语句

```python
# 推荐
with open("file.txt", "r") as f:
    content = f.read()
# 文件自动关闭

# 不推荐：忘记关闭
f = open("file.txt", "r")
content = f.read()
f.close()  # 容易忘记

# 紧急情况需要关闭
try:
    f = open("file.txt", "r")
    content = f.read()
finally:
    f.close()
```

## encoding 参数

```python
# 不同编码
with open("file.txt", "r", encoding="utf-8") as f:
    content = f.read()

with open("file.txt", "r", encoding="gbk") as f:
    content = f.read()

# 编码错误处理
with open("file.txt", "r", encoding="utf-8", errors="ignore") as f:
    content = f.read()

# errors 选项
errors="ignore"    # 忽略错误
errors="replace"   # 用 ? 替换
errors="strict"    # 默认，抛出异常
```

## 二进制文件

```python
# 读取图片
with open("image.png", "rb") as f:
    data = f.read()

# 写入图片
with open("copy.png", "wb") as f:
    f.write(data)

# 处理二进制数据
data = b"\x00\x01\x02\x03"
print(data[0])  # 0（int，不是 bytes）
```

## 文件指针

```python
with open("file.txt", "r+") as f:
    print(f.read())      # 读全部，指针在末尾
    f.write("end")       # 写在末尾

# 移动指针
with open("file.txt", "r") as f:
    f.seek(5)           # 移动到第 5 字节
    print(f.read(10))    # 读 10 个字节

# tell：当前位置
with open("file.txt", "r") as f:
    pos = f.tell()       # 0
    f.read(5)
    pos = f.tell()       # 5
```

## 读取大文件

```python
# 逐行读取（内存高效）
with open("huge_file.txt", "r") as f:
    for line in f:
        process(line)

# 固定大小分块
with open("large_file.bin", "rb") as f:
    while True:
        chunk = f.read(8192)  # 8KB 块
        if not chunk:
            break
        process(chunk)
```

## JSON 文件

```python
import json

# 写入 JSON
data = {"name": "Alice", "age": 30, "scores": [95, 88, 76]}
with open("data.json", "w", encoding="utf-8") as f:
    json.dump(data, f, ensure_ascii=False, indent=2)

# 读取 JSON
with open("data.json", "r", encoding="utf-8") as f:
    loaded = json.load(f)

# 字符串互转
json_str = json.dumps(data)
parsed = json.loads(json_str)
```

## CSV 文件

```python
import csv

# 写入 CSV
with open("output.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.writer(f)
    writer.writerow(["Name", "Age", "City"])
    writer.writerow(["Alice", 30, "NYC"])
    writer.writerow(["Bob", 25, "LA"])

# 读取 CSV
with open("output.csv", "r", newline="", encoding="utf-8") as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)

# 读取为字典
with open("output.csv", "r", newline="", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(row["Name"], row["Age"])
```

## os 模块文件操作

```python
import os

# 文件存在性
os.path.exists("file.txt")
os.path.isfile("file.txt")
os.path.isdir("folder")

# 目录操作
os.mkdir("new_folder")
os.makedirs("a/b/c")        # 递归创建
os.rmdir("folder")           # 删除空目录
os.remove("file.txt")       # 删除文件
os.rename("old.txt", "new.txt")

# 路径
os.path.join("folder", "file.txt")  # 跨平台路径
os.path.basename("/path/to/file.txt")  # "file.txt"
os.path.dirname("/path/to/file.txt")   # "/path/to"
os.path.split("/path/to/file.txt")     # ("/path/to", "file.txt")
```

## tempfile 临时文件

```python
import tempfile

# 临时文件（自动清理）
with tempfile.NamedTemporaryFile(mode='w', delete=True) as f:
    f.write("temp data")
    print(f.name)

# 临时目录
with tempfile.TemporaryDirectory() as tmpdir:
    print(f"临时目录: {tmpdir}")

# 临时文件路径
tmpfile = tempfile.mktemp(suffix=".txt")
print(f"临时文件: {tmpfile}")
```

## io 模块

```python
import io

# StringIO：内存中的文件对象
text_io = io.StringIO()
text_io.write("hello")
text_io.write(" world")
print(text_io.getvalue())  # "hello world"

# BytesIO：内存中的二进制文件对象
bytes_io = io.BytesIO()
bytes_io.write(b"\x00\x01\x02")
data = bytes_io.getvalue()

# 从现有文件读取
bytes_io = io.BytesIO(b"hello world")
text_io = io.TextIOWrapper(bytes_io, encoding="utf-8")
print(text_io.read())
```

## pathlib（现代路径操作）

```python
from pathlib import Path

# 创建路径对象
p = Path("folder/file.txt")

# 路径操作
p.parent          # Path("folder")
p.name            # "file.txt"
p.stem            # "file"
p.suffix          # ".txt"

# 存在性
p.exists()
p.is_file()
p.is_dir()

# 创建
p.mkdir(parents=True, exist_ok=True)
p.touch()  # 创建空文件

# 读取/写入
p.read_text(encoding="utf-8")
p.write_text("content", encoding="utf-8")
p.read_bytes()
p.write_bytes(b"binary")

# 遍历
for item in Path(".").iterdir():
    print(item.name)

# glob
for py_file in Path(".").glob("*.py"):
    print(py_file)
```

## 文件权限

```python
import os
import stat

# 获取权限
st = os.stat("file.txt")
oct(st.st_mode)  # '0o100644'

# 设置权限
os.chmod("file.txt", stat.S_IRUSR | stat.S_IWUSR)

# 检查
os.access("file.txt", os.R_OK)  # 可读
os.access("file.txt", os.W_OK)  # 可写
os.access("file.txt", os.X_OK)  # 可执行
```

## 小结

1. **open()**：文件操作核心，mode 参数决定读写/文本/二进制
2. **with 语句**：自动关闭文件，推荐使用
3. **encoding**：处理中文常用 utf-8
4. **read/readline/readlines**：不同读取方式
5. **seek/tell**：控制文件指针
6. **json/csv**：结构化数据存储
7. **pathlib**：现代 OO 路径操作
8. **tempfile**：临时文件/目录
9. **二进制读写**：用 `rb/wb` 模式
