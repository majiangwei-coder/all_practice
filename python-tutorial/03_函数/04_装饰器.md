# 装饰器

> 装饰器是 Python 的重要高级特性。本章详细讲解装饰器的原理、用法和常见模式。

---

## 1. 装饰器是什么

**装饰器**本质上是一个函数，**接受一个函数作为参数，返回一个新的函数**。

它的作用是：**在不修改原函数的前提下，给函数添加新的功能**。

```python
# 装饰器的本质：高阶函数
def my_decorator(func):  # func 是被装饰的函数
    def wrapper(*args, **kwargs):
        # 在调用前添加新功能
        print("调用前...")
        result = func(*args, **kwargs)  # 调用原函数
        # 在调用后添加新功能
        print("调用后...")
        return result
    return wrapper  # 返回包装后的函数
```

## 2. 装饰器的语法糖 @

### 不用 @ 的写法

```python
def greet():
    return "Hello!"

# 手动装饰
greet = my_decorator(greet)  # 等价于下面的 @ 语法

print(greet())  # 调用装饰后的函数
```

### 用 @ 的写法（推荐）

```python
@my_decorator
def greet():
    return "Hello!"

# 等价于：
# greet = my_decorator(greet)

print(greet())
```

## 3. 装饰器执行流程

让我们详细追踪装饰器的执行过程：

```python
# 1. 定义装饰器
def my_decorator(func):
    print(f"装饰器被调用，传入的函数: {func.__name__}")

    def wrapper(*args, **kwargs):
        print("执行前")
        result = func(*args, **kwargs)
        print("执行后")
        return result

    return wrapper

# 2. 用 @my_decorator 装饰函数
@my_decorator
def say_hello():
    print("Hello!")
    return "done"

# 3. 此时发生什么？
# say_hello = my_decorator(say_hello)
# my_decorator 被调用，打印 "装饰器被调用"
# 返回 wrapper 函数

# 4. 调用 say_hello() 时，实际调用的是 wrapper()
print("准备调用...")
result = say_hello()
print(f"返回值: {result}")
```

输出：

```
装饰器被调用，传入的函数: say_hello
准备调用...
执行前
Hello!
执行后
返回值: done
```

## 4. 装饰器带参数

如果装饰器需要参数，需要**返回装饰器的工厂函数**：

```python
# 外层：接收装饰器参数
def repeat(times):  # times 是装饰器参数
    print(f"创建装饰器，重复 {times} 次")

    # 中层：接收被装饰的函数
    def decorator(func):
        print(f"装饰函数: {func.__name__}")

        # 内层：包装函数
        def wrapper(*args, **kwargs):
            print(f"执行 {times} 次:")
            for i in range(times):
                func(*args, **kwargs)

        return wrapper

    return decorator  # 返回装饰器


# 使用
@repeat(times=3)
def say_hi():
    print("Hi!")

# 执行
say_hi()
```

输出：

```
创建装饰器，重复 3 次
装饰函数: say_hi
执行 3 次:
Hi!
Hi!
Hi!
```

### 图解执行顺序

```python
@repeat(times=3)  # 1. repeat(3) 被调用，返回 decorator
def say_hi():     # 2. decorator(say_hi) 被调用，返回 wrapper
    print("Hi!")

say_hi()  # 3. 调用 wrapper()
```

## 5. functools.wraps 保留元数据

装饰器会丢失原函数的信息，用 `functools.wraps` 解决：

```python
import functools

def my_decorator(func):
    @functools.wraps(func)  # 重要！保留原函数信息
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@my_decorator
def original():
    """这是 original 函数的文档"""
    pass

# 不使用 @functools.wraps：
# print(original.__name__)  # "wrapper"
# print(original.__doc__)  # None

# 使用后：
print(original.__name__)  # "original" ← 保留了！
print(original.__doc__)   # "这是 original 函数的文档"
```

## 6. 常见装饰器实现

### 6.1 计时装饰器

```python
import functools
import time

def timer(func):
    """测量函数执行时间"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()  # 高精度计时
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} 耗时 {elapsed:.4f} 秒")
        return result
    return wrapper


@timer
def slow_function():
    time.sleep(1)
    return "完成"


result = slow_function()
print(f"返回值: {result}")
```

### 6.2 日志装饰器

```python
import functools

def log(func):
    """记录函数调用日志"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f"调用 {func.__name__}({args}, {kwargs})")
        result = func(*args, **kwargs)
        print(f"{func.__name__} 返回 {result}")
        return result
    return wrapper


@log
def add(a, b):
    return a + b


result = add(3, 5)
```

### 6.3 参数验证装饰器

```python
import functools

def require_ints(func):
    """验证所有参数都是整数"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        for arg in args:
            if not isinstance(arg, int):
                raise TypeError(f"参数必须是整数，得到 {type(arg).__name__}")
        return func(*args, **kwargs)
    return wrapper


@require_ints
def add(a, b):
    return a + b


print(add(1, 2))     # 3
# add(1, "2")        # TypeError: 参数必须是整数
```

### 6.4 缓存装饰器 (memoization)

```python
import functools

def memoize(func):
    """缓存函数结果，避免重复计算"""
    cache = {}  # 存储结果的字典

    @functools.wraps(func)
    def wrapper(*args):
        if args not in cache:  # 如果没有缓存
            cache[args] = func(*args)  # 计算并缓存
        return cache[args]  # 返回缓存结果
    return wrapper


@memoize
def fibonacci(n):
    """斐波那契数列（递归版本）"""
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)


# 不带缓存：fibonacci(30) 需要非常久
# 带缓存：瞬间完成
print(fibonacci(30))  # 832040
```

## 7. 类装饰器

装饰器可以是类：

```python
class CallLogger:
    def __init__(self, func):
        # 保存原函数
        functools.update_wrapper(self, func)
        self.func = func

    def __call__(self, *args, **kwargs):
        """实例被调用时执行"""
        print(f"调用 {self.func.__name__}")
        return self.func(*args, **kwargs)


@CallLogger
def say_hello():
    print("Hello!")


say_hello()
# 输出：
# 调用 say_hello
# Hello!
```

## 8. 叠加多个装饰器

多个装饰器可以叠加，**执行顺序从近到远**：

```python
@decorator1  # 第3步执行
@decorator2  # 第2步执行
def greet():
    print("Hello!")

# 等价于：
# greet = decorator1(decorator2(greet))
```

### 执行顺序验证

```python
def decorator1(func):
    def wrapper(*args, **kwargs):
        print("1 start")
        result = func(*args, **kwargs)
        print("1 end")
        return result
    return wrapper

def decorator2(func):
    def wrapper(*args, **kwargs):
        print("2 start")
        result = func(*args, **kwargs)
        print("2 end")
        return result
    return wrapper

@decorator1
@decorator2
def greet():
    print("Hello!")
    return "done"


result = greet()
```

输出：

```
1 start
2 start
Hello!
2 end
1 end
返回值: done
```

### 图解叠加顺序

```
@greet 应用前
greet 函数本身

@greet 应用后（decorator2 先装饰）
greet = decorator2(greet)

再应用 decorator1
greet = decorator1(greet)

调用 greet() 时：
decorator1.wrapper → decorator2.wrapper → greet
```

## 9. 装饰器常见错误

### 错误1：忘记保存原函数信息

```python
# 错误写法
def bad_decorator(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@bad_decorator
def original():
    """文档"""
    pass

print(original.__name__)  # "wrapper" ← 原函数名丢失！
print(original.__doc__)   # None ← 原文档丢失！

# 正确写法
import functools

def good_decorator(func):
    @functools.wraps(func)  # 保留原函数信息
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper
```

### 错误2：装饰器带参数但少一层函数

```python
# 错误写法
def bad_timer(func):
    # 缺少一层函数接收 times 参数
    print("开始计时")
    return func  # 不是返回装饰器

# 正确写法
def good_timer(times):  # 第一层：接收参数
    def decorator(func):  # 第二层：接收函数
        @functools.wraps(func)
        def wrapper(*args, **kwargs):  # 第三层：包装
            # 计时逻辑
            return func(*args, **kwargs)
        return wrapper
    return decorator  # 返回装饰器
```

## 10. 更多实战装饰器

### 6.5 重试装饰器

```python
import functools
import time

def retry(max_attempts=3, delay=1, exceptions=(Exception,)):
    """失败自动重试"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    if attempt == max_attempts:
                        raise  # 最后一次还失败就抛出
                    print(f"第 {attempt} 次失败: {e}, {delay}秒后重试...")
                    time.sleep(delay)
            return None
        return wrapper
    return decorator


@retry(max_attempts=3, delay=1)
def unstable_network_call():
    import random
    if random.random() < 0.7:  # 70% 概率失败
        raise ConnectionError("网络异常")
    return "成功"
```

### 6.6 限流装饰器

```python
import functools
import time
from collections import deque

def rate_limit(max_calls=5, period=1):
    """限制函数调用频率"""
    calls = deque()

    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            now = time.time()
            # 清理过期的调用记录
            while calls and calls[0] < now - period:
                calls.popleft()
            if len(calls) >= max_calls:
                raise RuntimeError(f"调用太频繁！每秒最多 {max_calls} 次")
            calls.append(now)
            return func(*args, **kwargs)
        return wrapper
    return decorator

@rate_limit(max_calls=3, period=1)  # 每秒最多3次
def api_call(user_id):
    return f"查询用户 {user_id}"

# 快速调用会触发限制
for i in range(5):
    try:
        print(api_call(i))
    except RuntimeError as e:
        print(e)
```

### 6.7 注册表模式

```python
# 装饰器用于自动注册
handlers = {}

def register(command_name):
    """将函数注册为命令处理器"""
    def decorator(func):
        handlers[command_name] = func
        return func  # 返回原函数，不包装
    return decorator

@register("help")
def cmd_help():
    return "显示帮助信息"

@register("list")
def cmd_list():
    return "列出所有项目"

@register("quit")
def cmd_quit():
    return "退出程序"

# 使用
command = "help"
if command in handlers:
    print(handlers[command]())  # 显示帮助信息
```

## 12. 小结

### 必须掌握

1. **装饰器本质**：接受函数，返回包装后的函数
2. **@语法**：`@decorator` 等价于 `func = decorator(func)`
3. **执行顺序**：从下到上装饰，从上到下执行
4. **functools.wraps**：保留原函数元数据
5. **带参数装饰器**：需要三层函数（参数 → 函数 → wrapper）

### 常见错误

1. **忘记 @functools.wraps**：导致 `__name__` 和 `__doc__` 丢失
2. **装饰器顺序错误**：多个装饰器的顺序会影响执行顺序
3. **参数装饰器少一层**：带参数的装饰器需要三层嵌套
